\newglossaryentry{BaseDadosBibliografica}{
name = {Base de dados bibliográfica},
description  = {
Uma base de dados bibliográfica é um conjunto de registros bibliográficos, uma coleção organizada digital de referências à literaturas publicadas, incluindo revistas acadêmicas e artigos de jornais, proceedings de conferências, relatórios, publicações governamentais e legais, patentes, livros, etc.  
Fonte: \citet{wikipedia_base_2022}.
}
}
