\newglossaryentry{RC}{
	name={Revolução
Científica},
	description={As revoluções científicas são acontecimentos extraordinários nos quais ocorrem alterações de compromisso. Segundo Kuhn, é quando um paradigma é total ou parcialmente substituído por um novo, diferente do anterior, uma vez que a antiga não explica mais a realidade. Fonte: \citet{kuhn_estrutura_2001}}}


\newglossaryentry{dc}{
	name={Disciplina sintética},
	description={Processo de combinação do conhecimento existente com resultados de vários estudos de pesquisa de algum problema. Seus objetivos são tornar os resultados  dos estudos mais generalizáveis e aplicáveis, gerando novos conhecimentos para a tomada de decisões.
}}

\newglossaryentry{cg}{
	name={Cognição},
	description={Acúmulo e aquisição de informações pela aprendizagem ou experiência, que permite uma melhor adaptação ao meio.
}}
