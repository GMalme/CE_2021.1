\newglossaryentry{RegistroBibliografico}{
name = {Registro Bibliográfico},
description  = {
É um conjunto de dados ou palavras relacionadas, tratadas como um todo em termos físicos ou lógicos que servem para identificar um documento. São as informações bibliográficas tradicionalmente constantes dos catálogos de bibliotecas. Os registros são subdivididos em categorias de informação chamadas campos, tais como: título, autor e descritor.
Fonte: \cite{santos_gildenir_registro_2020}.
}
}
