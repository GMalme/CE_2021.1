\newglossaryentry{Dado}{
name = {dado},
description  = {
Dados são sequências de
fatos ainda não analisados, representativos de eventos que ocorrem nas organizações ou no ambiente
físico, antes de terem sido organizados e dispostos de forma que as pessoas possam entendê-los e usá-los.
Fonte: \cite{laudon_sistemas_2014}.
}
}

\newglossaryentry{Causalidade}{
name = {Causalidade},
description  = {
Condição segundo a qual uma causa produz um efeito.
Exemplo: O evento "derrubar um copo de vidro a uma altura de 20 metros" gerar o evento de "o copo se quebrou".
Fonte: \cite{dicio_causalidade_2022}
}
}
