\newglossaryentry{Experimentacao}{
	name={Experimentação},
	description={A experimentação é um processo de validação de técnicas, abordagens, ferramentas e teses, sendo importante em todas as áreas de conhecimento. Especificamente, na ciência da computação e na engenharia de software, onde estudos atuais quase sempre propõem abordagens, processos e produtos novos, o processo de experimentação deve ser parte fundamental do processo de criação de modelos. Fonte: \cite{marcolino_importancia_2014} }}
	
\newglossaryentry{ModelosCientificos}{
    name={Modelos Científicos},
    description={
    A modelagem científica é uma atividade científica, cujo objetivo é tornar uma parte ou característica
    específica do mundo mais fácil de entender, definir, quantificar, visualizar ou simular, referenciando-a ao conhecimento existente e geralmente aceito.}}
