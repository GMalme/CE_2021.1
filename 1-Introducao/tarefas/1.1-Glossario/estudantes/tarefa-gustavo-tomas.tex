\newglossaryentry{Positivismo}{
name = {Positivismo},
description = {
É uma corrente filosófica criada por Auguste Comte no século XIX que defende o conhecimento científico como única forma de conhecimento verdadeiro. Teorias apenas podem ser afirmadas por meio de comprovação pelo método empírico. Fonte: \cite{wikipedia_positivismo_2022}.
}
}
