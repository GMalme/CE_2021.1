\newglossaryentry{Empiricism}{
    name = {Empirismo Científico},
    description = {
    Empirismo científico é um princípio da produção científica baseado na filosofia positivista, cuja obtenção de conhecimento se dá pela observação ou análise da coleta de dados por meio da experimentação de um dado fenômeno.

    }
}
