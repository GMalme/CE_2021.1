\newglossaryentry{Cientista}{
	name={Cientista},
	description={
	Nome que se dá a pessoa que é responsável pela produção científica. Para atingir isso, ele faz uso do método científico, realizando pesquisas para obter conhecimento sobre determinado fato. Fonte: \citet{1.1-Ciencia} }
}

\newglossaryentry{planejamento}{
	name={planejamento},
	description={
	Ato ou efeito de planejar, criar um plano para otimizar a alcance de um determinado objetivo. Esta palavra pode abranger muitas áreas diferentes.
 }
}
