	name={Estatísticos},
description={Utilizam controles estatísticos e análise
de variáveis externas, são mais complexos, porém apresentam mais informações e permitem a avaliação de mais de uma variável ao mesmo tempo. Fonte: \cite{marcolino_importancia_2014} }}


\newglossaryentry{Heurísticas}{
	name={Heurísticas},
description={Heurísticas são técnicas desenvolvidas para resolver problemas mais rápido, quando os métodos clássicos são muito lentos para a busca das soluções apropriadas, ou quando os métodos tradicionais falham na busca de alguma solução. }}
